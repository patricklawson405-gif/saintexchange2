// auth-check.js - Include this on every page document.addEventListener('DOMContentLoaded', function() { updateAuthUI(); }); function updateAuthUI() { const userData = localStorage.getItem('saintUser'); const authContainer = document.getElementById('auth-buttons'); if (!authContainer) return; if (userData) { const user = JSON.parse(userData); // User is logged in - show user info + logout authContainer.innerHTML = `
Welcome, ${user.name}
`; } else { // User not logged in - show Login/Register authContainer.innerHTML = ` Login Register `; } } function logout() { localStorage.removeItem('saintUser'); window.location.reload(); // Refresh to update UI }
